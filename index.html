검증용 실제 쿼리 BigQuery
1. 노출, 옵션 선택, CTA 퍼널 (웹 이벤트)
-- 차종UP은 form_data에 co(원본 차종), cu(제안 차종) 키가 있어 cu 존재로 식별 SELECT CASE WHEN event_name='view' THEN '1_노출' WHEN component_name='car_upgrade_option' THEN '2_옵션선택' WHEN component_name='cta_button' THEN '3_CTA' END AS step, COUNT(DISTINCT member_id) AS uu FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND REGEXP_CONTAINS(form_data, r'cu') GROUP BY step -- 결과: 노출 1381, 옵션선택 500, CTA 1350 (전체 기간 distinct. 일자별 distinct 합과는 다름)
2. 결제 수락, 매출 (서버 결제)
-- paid_meta_info(meta=CAR_UPGRADE)와 paid_info를 id로 조인 WITH cu_paid AS ( SELECT pi.reservation_id, pi.amount, DATE(pmi.created_at,"Asia/Seoul") AS dt FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id WHERE REGEXP_CONTAINS(pmi.meta, r'CAR_UPGRADE') AND DATE(pmi.created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-22' ) SELECT dt, COUNT(DISTINCT reservation_id) AS accept, COUNT(DISTINCT reservation_id)*9900 AS customer_rev, SUM(amount) AS company_discount FROM cu_paid GROUP BY dt ORDER BY dt -- 결과: 6/10~6/22 결제 137건, 고객매출 1,356,300, 회사부담 2,462,850 (환불 0)
3. A/B 핵심 비교 (취소율 + 인당 결제액 + 객단가, 버킷팅 + 헤비 제외)
-- 버킷: farmhash(member_id + 'UPSELLING_CAR_UPGRADE') % 100 이 50 미만이면 노출군(B) -- (서버 코드 socar-common Testing.kt 의 ABTesting 공식. 노출자 1381명 전원 B로 검증됨) -- 법인/대량 계정(9일 20건 이상)은 A쪽에 쏠려 예약수를 왜곡하므로 제외 -- 대상 차급 = 차종UP 노출에서 실제 원본(co)으로 제안된 차급을 로그에서 직접 추출(하드코딩 없이 재현) -- 매출 = paid_info.amount 전 결제수단(카드/쿠폰/크레딧 등) 합 = 결제총액(GMV), 두 군 동일 정의 WITH co_list AS ( SELECT DISTINCT SAFE_CAST(REGEXP_EXTRACT(form_data, r'co\\?:\\?([0-9]+)') AS INT64) AS cid FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND page_name='upsell_bottomsheet' AND event_name='view' AND REGEXP_CONTAINS(form_data, r'cu') ), heavy AS ( SELECT member_id FROM `socar-data.tianjin_replica.reservation_info` WHERE DATE(created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-22' AND member_id > 0 GROUP BY member_id HAVING COUNT(*) >= 20 ), rsv AS ( SELECT ri.id AS reservation_id, ri.member_id, ri.cancel_at, MOD(ABS(FARM_FINGERPRINT(CAST(ri.member_id AS STRING)||'UPSELLING_CAR_UPGRADE')),100) < 50 AS is_b FROM `socar-data.tianjin_replica.reservation_info` ri WHERE DATE(ri.created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-22' AND ri.member_id > 0 AND ri.member_id NOT IN (SELECT member_id FROM heavy) AND ri.zone_id = ri.end_zone_id AND ri.car_class_id IN (SELECT cid FROM co_list WHERE cid IS NOT NULL) ), pay AS (SELECT reservation_id, SUM(amount) AS rev FROM `socar-data.tianjin_replica.paid_info` GROUP BY reservation_id) SELECT rsv.is_b, COUNT(*) AS rsv, COUNT(DISTINCT rsv.member_id) AS members, ROUND(COUNTIF(rsv.cancel_at IS NOT NULL)/COUNT(*)*100, 2) AS cancel_rate, ROUND(SUM(COALESCE(pay.rev,0))/COUNT(DISTINCT rsv.member_id)) AS arpu, ROUND(SUM(COALESCE(pay.rev,0))/COUNT(*)) AS rev_per_rsv FROM rsv LEFT JOIN pay USING(reservation_id) GROUP BY rsv.is_b -- 결과: A(대조) 43,061명/76,092예약 취소27.90% ARPU 219,063 -- B(노출) 42,824명/76,414예약 취소28.19% ARPU 219,169 -- 취소율 차이 없음(z~1.3), ARPU 거의 동일
4. 가입 면책상품별 차종UP 수락률
-- form_data pf = 진입 시점 적용 중인 면책상품(서버 protection_fee_info.deductible_type, 자기부담금 한도) -- form_data가 비이스케이프({pf:30..})와 이스케이프({\pf\:\30..}) 두 형태라 둘 다 잡는 정규식 사용 WITH exp AS ( SELECT member_id, REGEXP_EXTRACT(form_data, r'pf\\?:\\?([0-9A-Za-z_]+)') AS pf FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND page_name='upsell_bottomsheet' AND event_name='view' AND REGEXP_CONTAINS(form_data, r'cu') AND member_id > 0 ), exp1 AS (SELECT member_id, ANY_VALUE(pf) AS pf FROM exp GROUP BY member_id), acc AS ( SELECT DISTINCT ri.member_id FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'CAR_UPGRADE') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-10' AND '2026-06-22' ) SELECT e.pf, COUNT(DISTINCT e.member_id) AS exposed, COUNT(DISTINCT IF(a.member_id IS NOT NULL, e.member_id, NULL)) AS accepted FROM exp1 e LEFT JOIN acc a USING(member_id) GROUP BY e.pf ORDER BY exposed DESC -- 결과: 0_TOW(완전면책) 494명/43 8.7%, 30만원 400/41 10.3%, 70만원 458/31 6.8%, 5만원 29/6 20.7%(표본적음)
5. 제안 차종(차급)별 수락률
-- cu = 제안된 상위 차종의 carClass id. car_class에 조인해 차급(car_model)으로 묶음 WITH exp AS ( SELECT member_id, REGEXP_EXTRACT(form_data, r'cu\\?:\\?([0-9]+)') AS cu FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND page_name='upsell_bottomsheet' AND event_name='view' AND REGEXP_CONTAINS(form_data, r'cu') AND member_id > 0 ), exp1 AS (SELECT member_id, ANY_VALUE(cu) AS cu FROM exp GROUP BY member_id), acc AS ( SELECT DISTINCT ri.member_id FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'CAR_UPGRADE') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-10' AND '2026-06-22' ) SELECT cc.car_model AS car_class, COUNT(DISTINCT e.member_id) AS exposed, COUNT(DISTINCT IF(a.member_id IS NOT NULL, e.member_id, NULL)) AS accepted FROM exp1 e LEFT JOIN acc a USING(member_id) LEFT JOIN `socar-data.tianjin_replica.car_class` cc ON cc.id = SAFE_CAST(e.cu AS INT64) GROUP BY car_class ORDER BY exposed DESC -- 결과(상위 표본): 준대형 세단 210명/29 13.8%, 중형SUV 234/20 8.5%, 준중형SUV 892/65 7.3%

용어 정리 및 유의사항

노출 (view)
차종UP 바텀시트를 본 B군 회원 수. form_data에 co/cu 키가 있으면 차종UP 노출
옵션 선택
상위 차급 카드를 실제로 고른 사람(car_upgrade_option click). 원래 차종 유지는 로그가 남지 않음
결제 수락
차종UP을 결제 완료한 예약. 결제 meta upselling_type CAR_UPGRADE, 예약 단위 합계 양수만 집계(환불 제외)
수락률 단위
수락률은 수락 회원 122명 / 노출 회원 1381명 = 8.8%. 결제 건수와 매출은 건 단위(137건). 한 회원이 여러 번 결제할 수 있어 사람(122)과 건수(137)를 구분
고객 추가 매출
고객이 더 낸 금액. 9,900원 고정 x 수락 예약 수
회사 할인 부담 (참고)
상위 차급 차액 중 회사가 부담한 금액(paid_info의 upselling_discount 결제행 amount). 공헌이익 관점에서 상위 차종도 유휴였다면 손해로 보기 어려워 전면 지표로 두지 않음
예약 전환 가드레일
노출이 기존 예약을 깨는지 B 노출군과 A 대조군의 예약 취소율로 비교. 현재 A 27.90% vs B 28.19%로 사실상 차이 없음. 예약 수와 취소 수는 시간에 따라 변동하는 reservation 데이터라 재집계 시점마다 소폭 달라질 수 있음
CTA vs 옵션 선택
cta_button은 원래 차종으로 결제해도 찍히므로 수락 지표로 쓰지 않음. 실제 수락은 결제 meta로 판정
데이터 범위
차종UP 오픈 2026-06-10(CR-10867). 9일치 초기 데이터라 표본은 아직 작음
