SELECT CASE WHEN event_name='view' THEN '1_노출' WHEN component_name='car_upgrade_option' THEN '2_옵션선택' WHEN component_name='cta_button' THEN '3_CTA' END AS step, COUNT(DISTINCT member_id) AS uu FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND REGEXP_CONTAINS(form_data, r'cu') GROUP BY step -- 결과: 노출 1381, 옵션선택 500, CTA 1350
2. 결제 수락과 매출 (서버 결제)

3. A/B 취소율과 인당 결제액 비교 (버킷팅, 헤비 제외)
-- 버킷: farmhash(member_id + 'UPSELLING_CAR_UPGRADE') % 100 < 50 이면 노출군(B) -- 법인과 대량 계정(9일 20건 이상)은 제외, 대상 차급은 로그의 원본 차종(co)에서 추출 WITH co_list AS ( SELECT DISTINCT SAFE_CAST(REGEXP_EXTRACT(form_data, r'co\\?:\\?([0-9]+)') AS INT64) AS cid FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND page_name='upsell_bottomsheet' AND event_name='view' AND REGEXP_CONTAINS(form_data, r'cu')), heavy AS ( SELECT member_id FROM `socar-data.tianjin_replica.reservation_info` WHERE DATE(created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-22' AND member_id > 0 GROUP BY member_id HAVING COUNT(*) >= 20), rsv AS ( SELECT ri.id AS reservation_id, ri.member_id, ri.cancel_at, MOD(ABS(FARM_FINGERPRINT(CAST(ri.member_id AS STRING)||'UPSELLING_CAR_UPGRADE')),100) < 50 AS is_b FROM `socar-data.tianjin_replica.reservation_info` ri WHERE DATE(ri.created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-22' AND ri.member_id > 0 AND ri.member_id NOT IN (SELECT member_id FROM heavy) AND ri.zone_id = ri.end_zone_id AND ri.car_class_id IN (SELECT cid FROM co_list WHERE cid IS NOT NULL)) SELECT rsv.is_b, COUNT(*) AS rsv, COUNT(DISTINCT rsv.member_id) AS members, ROUND(COUNTIF(rsv.cancel_at IS NOT NULL)/COUNT(*)*100, 2) AS cancel_rate FROM rsv GROUP BY rsv.is_b -- 결과: A 76,106예약 취소 28.02% / B 76,468예약 취소 28.39% (z~1.6, 차이 없음), ARPU 양쪽 약 21.9만원
4. CS 문의율 A/B 비교
-- rsv는 위 3번과 동일. 예약 단위로 문의 연결 WITH inq AS ( SELECT reservation_id, COUNT(*) AS n FROM `socar-data.online_customer_center.inquiry` WHERE DATE(created_at,"Asia/Seoul") BETWEEN '2026-06-10' AND '2026-06-30' AND reservation_id IS NOT NULL GROUP BY reservation_id) SELECT rsv.is_b, ROUND(COUNTIF(i.reservation_id IS NOT NULL)/COUNT(*)*100,2) AS inq_rate FROM rsv LEFT JOIN inq i USING(reservation_id) GROUP BY rsv.is_b -- 결과: A 1.67%(1269/76106), B 1.66%(1272/76468), 차종UP 수락 137건 중 문의 2건
5. 제안 차급별 수락률
WITH exp AS ( SELECT member_id, REGEXP_EXTRACT(form_data, r'cu\\?:\\?([0-9]+)') AS cu FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN '2026-06-10' AND '2026-06-22' AND page_name='upsell_bottomsheet' AND event_name='view' AND REGEXP_CONTAINS(form_data, r'cu') AND member_id > 0), exp1 AS (SELECT member_id, ANY_VALUE(cu) AS cu FROM exp GROUP BY member_id), acc AS ( SELECT DISTINCT ri.member_id FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'CAR_UPGRADE') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-10' AND '2026-06-22') SELECT cc.car_model AS car_class, COUNT(DISTINCT e.member_id) AS exposed, COUNT(DISTINCT IF(a.member_id IS NOT NULL, e.member_id, NULL)) AS accepted FROM exp1 e LEFT JOIN acc a USING(member_id) LEFT JOIN `socar-data.tianjin_replica.car_class` cc ON cc.id = SAFE_CAST(e.cu AS INT64) GROUP BY car_class ORDER BY exposed DESC -- 결과: 준대형세단 210/29 13.8%, 중형SUV 234/20 8.5%, 준중형SUV 892/65 7.3%
